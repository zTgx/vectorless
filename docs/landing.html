How It Works

Three simple steps to intelligent retrieval

1

Parse Document

Documents are segmented into sections based on structure and content.

2

Build Tree

A hierarchical tree is created with summaries generated for each node.

3

Query & Navigate

The LLM navigates the tree to find the most relevant content for your query.

Simple to Integrate

Get started in minutes with our HTTP API

rust 
// Create a document
let response = reqwest::Client::new()
    .post("http://localhost:8080/documents")
    .json(&serde_json::json!({"title": "My Document"}))
    .send()
    .await?;

let doc: CreateDocumentResponse = response.json().await?;

// Upload content for indexing
reqwest::Client::new()
    .post(format!("http://localhost:8080/documents/{}/content", doc.id))
    .json(&serde_json::json!({"content": document_content}))
    .send()
    .await?;

// Query your knowledge base
let response = reqwest::Client::new()
    .post("http://localhost:8080/query")
    .json(&serde_json::json!({"query": "What is the main point?"}))
    .send()
    .await?;

let result: QueryResponse = response.json().await?;
println!("Answer: {}", result.answer);
